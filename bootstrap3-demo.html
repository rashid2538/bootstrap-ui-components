Bootstrap 3 UI Components

1.) alert
Syntax :
$bs.alert( message, callback, type, title, buttonText );
Example :
$bs.alert( 'Custom message to show.', function() {}, 'success', 'Custom Title', 'CustomButton' );
1.) confirm
Syntax :
$bs.confirm( message, callback, type, title, buttons );
Example :
$bs.confirm( 'Are you sure to proceed?', function( clickedButtonNumber ) {
	$bs.alert( 'You clicked button number ' + clickedButtonNumber + '.' );
}, 'success', 'Really?', [ 'Button1', 'Button2', 'Button3' ] );
3.) prompt
Syntax :
$bs.prompt( message, callback, type, title, buttonText );
Example :
$bs.prompt( 'Custom message to show.', function( value ) {
	$bs.alert( 'You entered : ' + value );
}, 'success', 'Custom Title', 'CustomButton' );
4.) popup
Syntax :
$bs.popup( url, type, title, callback, backdrop );
Example :
$bs.popup( 'https://www.youtube.com/embed/hwpOYkvS_1Q', 'default', 'I Love My Son', function() {}, false );
5.) loader
Syntax :
$bs.loader.show( options );
Example :
$bs.loader.show({
	message : 'Work started ...',
	icon : true	//needs font awesome to work
});
setTimeout( $bs.loader.hide.bind( $bs.loader ), 3000 );
6.) progress
Syntax :
$bs.progress.show( options );
Example :
$bs.progress.start( 'danger' );
setTimeout( $bs.progress.end.bind( $bs.progress ), 5000 );
